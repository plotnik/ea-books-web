<div class="mermaid">
sequenceDiagram
    participant U as User code
    participant R as RunnableWithMessageHistory
    participant H as get_session_history
    participant S as InMemoryChatMessageHistory
    participant P as ChatPromptTemplate
    participant L as llm
    participant O as StrOutputParser

    U->>R: invoke({"input": msg}, config)
    R->>H: get_session_history("user_123")
    H->>S: load/create history
    S-->>R: prior messages
    R->>P: fill prompt with system + history + input
    P-->>R: prompt value
    R->>L: send prompt
    L-->>R: AI message
    R->>O: parse output
    O-->>R: string response
    R->>S: append human + AI messages
    R-->>U: final response
