Call LLama

question = st.text_input("Question")

if st.button(':thinking_face: &nbsp; Call LLama'):
  response = ollama.chat(model='llama3.2', messages=[
    {
      'role': 'user',
      'content': question,
    },
  ])

  st.write(response.message.content)
